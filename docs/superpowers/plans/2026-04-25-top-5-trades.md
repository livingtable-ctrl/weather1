# Top 5 Trades of the Day Feature (Live Only)

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add a "Today's Top 5 Picks" section to the web dashboard showing the 5 highest-edge live market opportunities with their win probability, suggested dollar bet (Kelly-sized from a user-supplied daily budget), and confidence rating. The user can enter how much they want to spend today; the bet sizes scale proportionally to that budget. **Live markets only — paper trades are excluded.**

**Architecture:**
- New API endpoint `GET /api/top-trades?daily_budget=200` in `web_app.py`
  - Reads the latest `data/signals_cache.json` (same data the signals page uses; cron populates it)
  - Filters to `signal` containing `"STRONG"` — only high-confidence picks
  - Sorts by `abs(edge_pct)` descending, returns top 5
  - For each pick: computes suggested bet = `(edge_pct/100 / sum_of_all_edges) * daily_budget` (proportional Kelly sizing)
  - Returns `[{ticker, city, side, edge_pct, win_prob, suggested_bet, stars, time_risk}]`
- New `GET /api/top-trades` also accepts `daily_budget` from a saved pref in `data/user_prefs.json` (same file as profit goal) as a fallback when no query param given
- New `POST /api/top-trades/budget` saves the daily budget to `user_prefs.json`
- Dashboard UI (`templates/dashboard.html` + `static/dashboard.js`): new card "Today's Picks" with a budget input and a table of top 5 trades

**Tech Stack:** Python (`web_app.py`), JavaScript (`static/dashboard.js`), HTML (`templates/dashboard.html`), `tests/test_top_trades.py` (new)

    @app.route("/api/top-trades", methods=["GET"])
    def api_top_trades():
        """Return top 5 strong trade picks with Kelly-proportional bet sizes."""
        try:
            daily_budget = float(request.args.get("daily_budget", 0) or 0)
            if daily_budget <= 0:
                # Fall back to saved pref
                daily_budget = _load_daily_budget()

            if not _SIGNALS_CACHE_PATH.exists():
                return jsonify({
                    "trades": [],
                    "daily_budget": daily_budget,
                    "message": "No scan data yet — run a cron scan first.",
                })

            with open(_SIGNALS_CACHE_PATH, encoding="utf-8") as f:
                cache = json.load(f)

            all_signals = cache.get("signals", [])
            # Live only: filter to STRONG signals
            strong = [s for s in all_signals if "STRONG" in str(s.get("signal", ""))]
            strong.sort(key=lambda s: abs(s.get("edge_pct", 0)), reverse=True)
            top5 = strong[:5]

            # Proportional Kelly bet sizing
            total_edge = sum(abs(s.get("edge_pct", 0)) for s in top5) or 1.0
            trades = []
            for s in top5:
                edge = abs(s.get("edge_pct", 0))
                suggested = round((edge / total_edge) * daily_budget, 2) if daily_budget > 0 else 0.0
                trades.append({
                    "ticker": s.get("ticker", ""),
                    "city": s.get("city", "—"),
                    "side": s.get("side", "—"),
                    "edge_pct": s.get("edge_pct", 0),
                    "win_prob": s.get("forecast_prob", 0),
                    "suggested_bet": suggested,
                    "stars": s.get("stars", 0),
                    "time_risk": s.get("time_risk", "?"),
                    "near_threshold": s.get("near_threshold", False),
                })

            return jsonify({
                "trades": trades,
                "daily_budget": daily_budget,
                "generated_at": cache.get("generated_at"),
            })
        except Exception as exc:
            return jsonify({"error": str(exc), "trades": []}), 500

        if _PREFS_PATH.exists():

    def api_set_daily_budget():
        try:
            body = request.get_json(force=True) or {}
            budget = float(body.get("daily_budget", 0))
        except (TypeError, ValueError):
            return jsonify({"error": "daily_budget must be a number"}), 400
        if budget <= 0:
            return jsonify({"error": "daily_budget must be positive"}), 400
        # Reuse the same prefs file as profit goal
        _PREFS_PATH.parent.mkdir(parents=True, exist_ok=True)
        try:
            existing: dict = {}
            if _PREFS_PATH.exists():
                existing = _json.loads(_PREFS_PATH.read_text())
        except Exception:
            existing = {}
        existing["daily_budget"] = budget
        _PREFS_PATH.write_text(_json.dumps(existing, indent=2))
        return jsonify({"daily_budget": budget})

```javascript
    function loadTopTrades(budget) {
      var url = '/api/top-trades' + (budget ? '?daily_budget=' + budget : '');
      fetch(url)
        .then(function(r) { return r.json(); })
        .then(function(d) {
          var el = document.getElementById('top-trades-table');
          if (!el) return;
          var trades = d.trades || [];
          if (!trades.length) {
            el.innerHTML = '<p style="color:var(--text-muted);font-size:12px">' +
              (d.message || 'No strong signals available. Run a cron scan to refresh.') + '</p>';
            return;
          }
          var table = document.createElement('table');
          table.style.width = '100%';
          table.style.fontSize = '12px';
          table.style.borderCollapse = 'collapse';
          var thead = table.createTHead();
          thead.innerHTML = '<tr><th>Ticker</th><th>City</th><th>Side</th>' +
            '<th>Edge</th><th>Win%</th><th>Suggested Bet</th><th>Risk</th></tr>';
          var tbody = table.createTBody();
          trades.forEach(function(t) {
            var row = tbody.insertRow();
            row.insertCell().textContent = t.ticker;
            row.insertCell().textContent = t.city;
            var sideCell = row.insertCell();
            sideCell.textContent = t.side;
            sideCell.style.color = t.side === 'YES' ? 'var(--pos)' : 'var(--neg)';
            row.insertCell().textContent = (t.edge_pct || 0).toFixed(1) + '%';
            row.insertCell().textContent = t.win_prob ? (t.win_prob * 100).toFixed(0) + '%' : '—';
            var betCell = row.insertCell();
            betCell.textContent = t.suggested_bet > 0 ? '$' + t.suggested_bet.toFixed(2) : '—';
            betCell.style.fontWeight = 'bold';
            var riskCell = row.insertCell();
            riskCell.textContent = t.time_risk || '?';
            riskCell.style.color = t.time_risk === 'LOW' ? 'var(--pos)' :
                                   t.time_risk === 'HIGH' ? 'var(--neg)' : 'var(--warn)';
          });
          el.innerHTML = '';
          el.appendChild(table);
          // Update budget input with the budget that was used
          var inp = document.getElementById('daily-budget-input');
          if (inp && !inp.value && d.daily_budget) inp.value = d.daily_budget;
        })
        .catch(function() {
          var el = document.getElementById('top-trades-table');
          if (el) el.innerHTML = '<p style="color:var(--neg);font-size:12px">Failed to load picks.</p>';
        });
    }

    window.applyDailyBudget = function() {
      var inp = document.getElementById('daily-budget-input');
      var budget = inp ? parseFloat(inp.value) : 0;
      if (!budget || budget <= 0) { alert('Enter a positive budget amount.'); return; }
      // Save pref
      fetch('/api/top-trades/budget', {
        method: 'POST',
        headers: {'Content-Type': 'application/json'},
        body: JSON.stringify({daily_budget: budget})
      }).then(function() { loadTopTrades(budget); });
    };

    loadTopTrades(0);  // 0 = use saved pref
```
